# Filter out non-numeric columns
numeric_processors <- processors[sapply(processors, is.numeric)]

# Compute the mean, variance, minimum, and maximum
mean_values <- colMeans(numeric_processors, na.rm = TRUE)
variance_values <- apply(numeric_processors, 2, var, na.rm = TRUE)
min_values <- apply(numeric_processors, 2, min, na.rm = TRUE)
max_values <- apply(numeric_processors, 2, max, na.rm = TRUE)

# Sort the columns
sorted_processors <- processors
sorted_processors[] <- lapply(numeric_processors, function(x) sort(x, na.last = TRUE))

# Compute the fraction of NA values
na_fraction <- colMeans(is.na(numeric_processors))


cat("Mean Values:\n")
print(mean_values)
cat("\nVariance Values:\n")
print(variance_values)
cat("\nMinimum Values:\n")
print(min_values)
cat("\nMaximum Values:\n")
print(max_values)
cat("\nSorted Processors Data:\n")
print(sorted_processors)
cat("\nFraction of NA Values:\n")
print(na_fraction)
